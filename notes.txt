#!/bin/bash

read -p "enter number " num

count=0

if [ $num -eq 0 ] || [ $num -eq 1 ]

then echo "not a prime no"

elif [ $num -eq 2 ]

then echo "prime no"

else

for i in { 1..$num..1 }

do

if [ `expr $num % $i` == 0 ]

then 
	count = `expr $count + 1`

fi 

done

if [ $count -eq 2 ]
then echo "prime no"
else
echo "not a prime no"
fi
fi
